CREATE TABLE STAGE (
    CODE_STAGE CHAR(5) PRIMARY KEY,
    TYPE_STAGE CHAR(5),
    DEBUT_STAGE DATE,
    FIN_STAGE DATE,
    NBPLACE_STAGE NUMBER(2),
    NBINSCRIT_STAGE NUMBER(2)
);

CREATE TABLE STAGIAIRE (
    NUM_STAGIAIRE CHAR(5) PRIMARY KEY,
    NOM_STAGIAIRE CHAR(15),
    PRENOM_STAGIAIRE CHAR(15),
    SEXE_STAGIAIRE CHAR(1),
    DNAISS_STAGIAIRE DATE,
    DIPLO_STAGIAIRE CHAR(3)
);

CREATE TABLE INSCRIPTION (
    CODE_STAGE CHAR(5),
    NUM_STAGIAIRE CHAR(5),
    DATE_INSCRIP DATE,
    STATUT_INSCRIP CHAR(1),
    CODE_POSITION CHAR(1),
    PRIMARY KEY (CODE_STAGE, NUM_STAGIAIRE),
    FOREIGN KEY (CODE_STAGE) REFERENCES STAGE(CODE_STAGE),
    FOREIGN KEY (NUM_STAGIAIRE) REFERENCES STAGIAIRE(NUM_STAGIAIRE)
);
